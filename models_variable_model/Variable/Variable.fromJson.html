factory Variable.fromJson(JSON json) {
  Map<String, dynamic> _estacion = json["Estacion"] as Map<String, dynamic>;
  Map<String, dynamic> _horario = json["Horario"] as Map<String, dynamic>;
  Map<String, dynamic> _var = json["Variable"] as Map<String, dynamic>;
  Map<String, dynamic>? _instrumento = json["Instrumento"];

  return Variable(
      id: json["id"],
      estacion: _estacion["codigo"],
      tipoHora: _horario["tipoHora"],
      hora: _horario["hora"],
      nombre: _var["nombre"],
      unidad: _var["unidad"],
      max: (_var["maximo"] != null)
          ? (_var["maximo"] as int).toDouble()
          : null,

          : null,

      descripcion: _var["descripcion"],
      instrumento: (_instrumento != null) ? _instrumento["nombre"] : null);
}
